新增 链表功能
<- 为 prev
-> 为 next
依次想空链表中插入 new_0、new_1、new_2

头插法
head <-> new_2 <-> new_1 <-> new_0 <-> head

尾插法
head <-> new_0 <-> new_1 <-> new_2 <-> head

static _EF_INLINE void ef_list_add(ef_list_t *p_new, ef_list_t *p_head)
{
    __ef_list_add(p_new, p_head, p_head->next);
}

static _EF_INLINE void ef_list_add_tail(ef_list_t *p_new, ef_list_t *p_head)
{
    __ef_list_add(p_new, p_head->prev, p_head);
}
